CREATE DATABASE sathyabama;
USE sathyabama;
CREATE TABLE Category (
    id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    dno INT NOT NULL
);
CREATE TABLE Product (
    id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    dno INT NOT NULL,
    category_id INT NOT NULL,
    FOREIGN KEY (category_id) REFERENCES Category(id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
CREATE TABLE Orders (
    id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    age INT NOT NULL,
    product_id INT NOT NULL,
    FOREIGN KEY (product_id) REFERENCES Product(id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

CREATE VIEW StudentProducts AS
SELECT C.name AS Student_Name, P.name AS Product_Name, P.dno
FROM Category C
JOIN Product P ON C.id = P.category_id;
SELECT * FROM StudentProducts;
CREATE VIEW ProductOrders AS
SELECT P.name AS Product_Name, O.name AS Order_Name, O.age
FROM Product P
JOIN Orders O ON P.id = O.product_id;
SELECT * FROM ProductOrders;
